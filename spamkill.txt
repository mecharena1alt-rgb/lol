-- Services
local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local CoreGui = game:GetService("CoreGui")

-- Player and Character
local player = Players.LocalPlayer

-- UI Construction
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "SpamKillGui"
screenGui.Parent = CoreGui
screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

local mainFrame = Instance.new("Frame")
mainFrame.Name = "MainFrame"
mainFrame.Size = UDim2.new(0, 180, 0, 60)
mainFrame.Position = UDim2.new(0.05, 0, 0.5, -30)
mainFrame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
mainFrame.BorderSizePixel = 0
mainFrame.Active = true
mainFrame.Draggable = true
mainFrame.Parent = screenGui

local uiCorner = Instance.new("UICorner")
uiCorner.CornerRadius = UDim.new(0, 8)
uiCorner.Parent = mainFrame

local titleLabel = Instance.new("TextLabel")
titleLabel.Name = "TitleLabel"
titleLabel.Size = UDim2.new(1, 0, 0, 25)
titleLabel.Position = UDim2.new(0, 0, 0, 0)
titleLabel.BackgroundTransparency = 1
titleLabel.Text = "Spam Kill"
titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
titleLabel.Font = Enum.Font.GothamBold
titleLabel.TextSize = 14
titleLabel.Parent = mainFrame

local toggleBtn = Instance.new("TextButton")
toggleBtn.Name = "ToggleBtn"
toggleBtn.Size = UDim2.new(1, -10, 0, 25)
toggleBtn.Position = UDim2.new(0, 5, 0, 30)
toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50) -- Red (Off)
toggleBtn.Text = "SPAM KILL: OFF"
toggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
toggleBtn.Font = Enum.Font.Gotham
toggleBtn.TextSize = 13
toggleBtn.Parent = mainFrame

local btnCorner = Instance.new("UICorner")
btnCorner.CornerRadius = UDim.new(0, 6)
btnCorner.Parent = toggleBtn

-- State
local isEnabled = false
local spamKillConnection = nil

-- Spam Kill Logic
local function spamKill()
    local enemies = Players:GetPlayers()
    for _, enemy in ipairs(enemies) do
        if enemy ~= player and enemy.Character and enemy.Character:FindFirstChild("HumanoidRootPart") then
            game:GetService("ReplicatedStorage").Remotes.ShootGun:FireServer(, , enemy.Character.HumanoidRootPart)
        end
    end
end

local function startSpamKill()
    if spamKillConnection then return end

    spamKillConnection = RunService.Heartbeat:Connect(function()
        if not isEnabled then
            return -- Stop if disabled
        end

        spamKill()
    end)
end

local function stopSpamKill()
    if spamKillConnection then
        spamKillConnection:Disconnect()
        spamKillConnection = nil
    end
end

-- Toggle Logic
toggleBtn.MouseButton1Click:Connect(function()
    isEnabled = not isEnabled

    if isEnabled then
        toggleBtn.Text = "SPAM KILL: ON"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 200, 50) -- Green (On)
        startSpamKill()
    else
        toggleBtn.Text = "SPAM KILL: OFF"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50) -- Red (Off)
        stopSpamKill()
    end
end)

-- Cleanup (Press RightControl to close the UI)
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if not gameProcessed and input.KeyCode == Enum.KeyCode.P then
        stopSpamKill()
        if screenGui then
            screenGui:Destroy()
        end
    end
end)

print("Spam Kill UI loaded. Press RightControl to close.")
