-- state
local isEnabled = false
local currentMatchData = nil

-- ENEMY LOGIC ---------------------------------------------------------
local function isEnemy(playerToCheck)
    -- If MatchService failed or no data, fallback to "everyone is enemy" (safer for testing)
    -- or return false if you want strict safety.
    if not currentMatchData then return true end
    
    -- Don't target self
    if playerToCheck == LocalPlayer then return false end

    local myData = currentMatchData[LocalPlayer]
    local enemyData = currentMatchData[playerToCheck]

    -- If we can't find team data, ignore them
    if not myData or not enemyData then return false end

    -- The actual check: are teams different?
    return enemyData.Team ~= myData.Team
end

-- Hook up the signal to get match data
if MatchService then
    MatchService.GetClientMatchStartedSignal():Connect(function(_, matchData)
        currentMatchData = matchData
        print("Match started, enemy data updated.")
    end)
end
------------------------------------------------------------------------

local function getVectorSize(inputStr)
    local num = tonumber(inputStr)
    if num then
        return Vector3.new(num, num, num)
    end
    return Vector3.new(5, 5, 5) 
end

local function addOutline(part)
    if not part:FindFirstChild("ESPBox") then
        local box = Instance.new("BoxHandleAdornment")
        box.Name = "ESPBox"
        box.Adornee = part
        box.AlwaysOnTop = true 
        box.ZIndex = 5
        box.Size = part.Size
        box.Transparency = 0.5
        box.Color3 = Color3.fromRGB(255, 50, 50) -- Red for enemies
        box.Parent = part
    else
        part.ESPBox.Size = part.Size
    end
end

local function removeOutline(part)
    if part:FindFirstChild("ESPBox") then
        part.ESPBox:Destroy()
    end
end

local function expandHitboxes()
    for _, player in pairs(Players:GetPlayers()) do
        -- ONLY run on enemies
        if isEnemy(player) and player.Character then
            local char = player.Character
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChild("Humanoid")

            if hrp and hum and hum.Health > 0 then
                -- Target specifically the "Part" inside HRP, fallback to HRP
                local target = hrp:FindFirstChild("Part") or hrp
                
                target.Size = getVectorSize(sizeInput.Text)
                target.CanCollide = false
                addOutline(target)
            end
        end
    end
end

local function resetHitboxes()
    for _, player in pairs(Players:GetPlayers()) do
        if player.Character then
            local char = player.Character
            local hrp = char:FindFirstChild("HumanoidRootPart")
            
            if hrp then
                local target = hrp:FindFirstChild("Part") or hrp
                target.Size = Vector3.new(1, 1, 1) -- Standard reset
                removeOutline(target)
            end
        end
    end
end

-- loop
RunService.RenderStepped:Connect(function()
    if isEnabled then
        expandHitboxes()
    end
end)

-- toggle logic
toggleBtn.MouseButton1Click:Connect(function()
    isEnabled = not isEnabled
    if isEnabled then
        toggleBtn.Text = "EXPANDER: ON"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 200, 50)
    else
        toggleBtn.Text = "EXPANDER: OFF"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
        resetHitboxes()
    end
end)

-- cleanup
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if not gameProcessed and input.KeyCode == Enum.KeyCode.P then
        isEnabled = false
        if screenGui then screenGui:Destroy() end
    end
end)
