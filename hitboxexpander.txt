-- state
local isEnabled = false

-- logic
local function getVectorSize(inputStr)
    -- tonumber handles decimals automatically (e.g. "3.3" becomes 3.3)
    local num = tonumber(inputStr)
    if num then
        return Vector3.new(num, num, num)
    end
    return Vector3.new(5, 5, 5) -- fallback
end

local function addOutline(part)
    if not part:FindFirstChild("ESPBox") then
        local box = Instance.new("BoxHandleAdornment")
        box.Name = "ESPBox"
        box.Adornee = part
        box.AlwaysOnTop = true -- this is the x-ray magic
        box.ZIndex = 5
        box.Size = part.Size
        box.Transparency = 0.5
        box.Color3 = Color3.fromRGB(200, 200, 200) -- whitish grey
        box.Parent = part
    else
        -- update size if it changes
        part.ESPBox.Size = part.Size
    end
end

local function removeOutline(part)
    if part:FindFirstChild("ESPBox") then
        part.ESPBox:Destroy()
    end
end

local function expandHitboxes()
    for _, obj in pairs(workspace:GetDescendants()) do
        -- looking for HumanoidRootPart
        if obj.Name == "HumanoidRootPart" and obj.Parent:FindFirstChild("Humanoid") then
            local char = obj.Parent
            
            if char.Name ~= player.Name then
                -- Target specifically the "Part" inside HRP, fallback to HRP if missing
                local target = obj:FindFirstChild("Part") or obj
                
                target.Size = getVectorSize(sizeInput.Text)
                target.CanCollide = false
                -- We do NOT change transparency, we add the outline instead
                addOutline(target)
            end
        end
    end
end

local function resetHitboxes()
    for _, obj in pairs(workspace:GetDescendants()) do
        if obj.Name == "HumanoidRootPart" and obj.Parent:FindFirstChild("Humanoid") then
             local char = obj.Parent
             if char.Name ~= player.Name then
                local target = obj:FindFirstChild("Part") or obj
                -- standard size reset. if the game uses custom sizes this might look weird but it works
                target.Size = Vector3.new(1, 1, 1) 
                removeOutline(target)
             end
        end
    end
end

-- loop
runService.RenderStepped:Connect(function()
    if isEnabled then
        expandHitboxes()
    end
end)

-- toggle logic
toggleBtn.MouseButton1Click:Connect(function()
    isEnabled = not isEnabled
    if isEnabled then
        toggleBtn.Text = "EXPANDER: ON"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 200, 50)
    else
        toggleBtn.Text = "EXPANDER: OFF"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
        resetHitboxes()
    end
end)

-- cleanup
userInputService.InputBegan:Connect(function(input, gameProcessed)
    if not gameProcessed and input.KeyCode == Enum.KeyCode.P then
        isEnabled = false
        if screenGui then screenGui:Destroy() end
    end
end)
