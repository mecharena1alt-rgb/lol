-- Services
local Players = game:GetService("Players")
local Workspace = game:GetService("Workspace")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local CoreGui = game:GetService("CoreGui")

-- Player and Character
local player = Players.LocalPlayer

-- Config
local weaponName = "ShroudTool" -- Change this to your desired weapon name

-- UI Construction
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "ShroudBypass"
screenGui.Parent = CoreGui
screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

local mainFrame = Instance.new("Frame")
mainFrame.Name = "MainFrame"
mainFrame.Size = UDim2.new(0, 180, 0, 60)
mainFrame.Position = UDim2.new(0.05, 0, 0.5, -30)
mainFrame.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
mainFrame.BorderSizePixel = 0
mainFrame.Active = true
mainFrame.Draggable = true
mainFrame.Parent = screenGui

local uiCorner = Instance.new("UICorner")
uiCorner.CornerRadius = UDim.new(0, 8)
uiCorner.Parent = mainFrame

local titleLabel = Instance.new("TextLabel")
titleLabel.Name = "TitleLabel"
titleLabel.Size = UDim2.new(1, 0, 0, 25)
titleLabel.Position = UDim2.new(0, 0, 0, 0)
titleLabel.BackgroundTransparency = 1
titleLabel.Text = "Shroud Bypass"
titleLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
titleLabel.Font = Enum.Font.GothamBold
titleLabel.TextSize = 14
titleLabel.Parent = mainFrame

local toggleBtn = Instance.new("TextButton")
toggleBtn.Name = "ToggleBtn"
toggleBtn.Size = UDim2.new(1, -10, 0, 25)
toggleBtn.Position = UDim2.new(0, 5, 0, 30)
toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50) -- Red (Off)
toggleBtn.Text = "BYPASS: OFF"
toggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
toggleBtn.Font = Enum.Font.Gotham
toggleBtn.TextSize = 13
toggleBtn.Parent = mainFrame

local btnCorner = Instance.new("UICorner")
btnCorner.CornerRadius = UDim.new(0, 6)
btnCorner.Parent = toggleBtn

-- State
local isEnabled = false
local equipConnection = nil

-- Logic
local function equipWeapon()
    -- Ensure character and backpack exist
    local character = player.Character
    local backpack = player:FindFirstChild("Backpack")
    
    if not character or not backpack then return end

    -- Find the tool in the backpack
    local tool = backpack:FindFirstChild(weaponName)
    
    if tool then
        -- Move the tool from the Backpack to the Character
        tool.Parent = character
        print("Equipped '" .. weaponName .. "'.")
    end
end

local function startEquipLoop()
    if equipConnection then return end -- Prevent multiple loops
    
    -- Initial equip attempt
    equipWeapon()
    
    -- Loop to keep re-equipping if it gets removed
    equipConnection = RunService.Heartbeat:Connect(function()
        if not isEnabled or not player.Character then
            return -- Stop if disabled or character is gone
        end
        
        local backpack = player:FindFirstChild("Backpack")
        if not backpack then return end
        
        -- Check if the weapon is in the backpack but not in the character
        if backpack:FindFirstChild(weaponName) and not player.Character:FindFirstChild(weaponName) then
            equipWeapon()
        end
    end)
end

local function stopEquipLoop()
    if equipConnection then
        equipConnection:Disconnect()
        equipConnection = nil
    end
end

-- Re-connect on respawn to ensure the loop keeps working
player.CharacterAdded:Connect(function()
    print("Character respawned. Re-checking bypass state.")
    if isEnabled then
        -- A small delay to let the character and backpack load
        task.delay(1, startEquipLoop)
    end
end)


-- Toggle Logic
toggleBtn.MouseButton1Click:Connect(function()
    isEnabled = not isEnabled
    
    if isEnabled then
        toggleBtn.Text = "BYPASS: ON"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(50, 200, 50) -- Green (On)
        startEquipLoop()
    else
        toggleBtn.Text = "BYPASS: OFF"
        toggleBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 50) -- Red (Off)
        stopEquipLoop()
    end
end)

-- Cleanup (Press RightControl to close the UI)
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if not gameProcessed and input.KeyCode == Enum.KeyCode.P then
        stopEquipLoop()
        if screenGui then
            screenGui:Destroy()
        end
    end
end)

print("Weapon Equip Bypass UI loaded. Press RightControl to close.")
